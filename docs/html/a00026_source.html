59Error_t CCombFilterBase::setParam( CCombFilterIf::FilterParam_t eParam, float fParamValue )
60{
61 if (!isInParamRange(eParam, fParamValue))
62 return Error_t::kFunctionInvalidArgsError;
63
64 // special actions for special parameters
65 if (eParam == CCombFilterIf::kParamDelay)
66 {
67 int iNumAdditionalTaps = CUtil::float2int<int>(fParamValue - m_afParam[CCombFilterIf::kParamDelay]);
68 if (iNumAdditionalTaps < 0)
69 {
70 for (int c = 0; c < m_iNumChannels; c++)
71 {
72 m_ppCRingBuffer[c]->setWriteIdx(CUtil::float2int<int>(fParamValue) + m_ppCRingBuffer[c]->getReadIdx());
73 }
74 }
75 else
76 {
77
78 for (int c = 0; c < m_iNumChannels; c++)
79 {
80 for (int i = 0; i < iNumAdditionalTaps; i++)
81 {
82 m_ppCRingBuffer[c]->putPostInc(0.F);
83 }
84 }
85 }
86 }
87
88 m_afParam[eParam] = fParamValue;
89
90 return Error_t::kNoError;
91}
