582bool MatchPrintAndExplain(Value& value, const Matcher<T>& matcher,
583 MatchResultListener* listener) {
584 if (!listener->IsInterested()) {
585 // If the listener is not interested, we do not need to construct the
586 // inner explanation.
587 return matcher.Matches(value);
588 }
589
590 StringMatchResultListener inner_listener;
591 const bool match = matcher.MatchAndExplain(value, &inner_listener);
592
593 UniversalPrint(value, listener->stream());
594#if GTEST_HAS_RTTI
595 const std::string& type_name = GetTypeName<Value>();
596 if (IsReadableTypeName(type_name))
597 *listener->stream() << " (of type " << type_name << ")";
598#endif
599 PrintIfNotEmpty(inner_listener.str(), listener->stream());
600
601 return match;
602}

604// An internal helper class for doing compile-time loop on a tuple's
605// fields.
606template <size_t N>
607class TuplePrefix {
608 public:
609 // TuplePrefix<N>::Matches(matcher_tuple, value_tuple) returns true
610 // if and only if the first N fields of matcher_tuple matches
611 // the first N fields of value_tuple, respectively.
612 template <typename MatcherTuple, typename ValueTuple>
613 static bool Matches(const MatcherTuple& matcher_tuple,
614 const ValueTuple& value_tuple) {
615 return TuplePrefix<N - 1>::Matches(matcher_tuple, value_tuple) &&
616 std::get<N - 1>(matcher_tuple).Matches(std::get<N - 1>(value_tuple));
617 }
618
619 // TuplePrefix<N>::ExplainMatchFailuresTo(matchers, values, os)
620 // describes failures in matching the first N fields of matchers
621 // against the first N fields of values. If there is no failure,
622 // nothing will be streamed to os.
623 template <typename MatcherTuple, typename ValueTuple>
624 static void ExplainMatchFailuresTo(const MatcherTuple& matchers,
625 const ValueTuple& values,
626 ::std::ostream* os) {
627 // First, describes failures in the first N - 1 fields.
628 TuplePrefix<N - 1>::ExplainMatchFailuresTo(matchers, values, os);
629
630 // Then describes the failure (if any) in the (N - 1)-th (0-based)
631 // field.
632 typename std::tuple_element<N - 1, MatcherTuple>::type matcher =
633 std::get<N - 1>(matchers);
634 typedef typename std::tuple_element<N - 1, ValueTuple>::type Value;
635 const Value& value = std::get<N - 1>(values);
636 StringMatchResultListener listener;
637 if (!matcher.MatchAndExplain(value, &listener)) {
638 *os << " Expected arg #" << N - 1 << ": ";
639 std::get<N - 1>(matchers).DescribeTo(os);
640 *os << "\n Actual: ";
641 // We remove the reference in type Value to prevent the
642 // universal printer from printing the address of value, which
643 // isn't interesting to the user most of the time. The
644 // matcher's MatchAndExplain() method handles the case when
645 // the address is interesting.
646 internal::UniversalPrint(value, os);
647 PrintIfNotEmpty(listener.str(), os);
648 *os << "\n";
649 }
650 }
651};

696// TransformTupleValues and its helper.
697//
698// TransformTupleValuesHelper hides the internal machinery that
699// TransformTupleValues uses to implement a tuple traversal.
700template <typename Tuple, typename Func, typename OutIter>
701class TransformTupleValuesHelper {
702 private:
703 typedef ::std::tuple_size<Tuple> TupleSize;
704
705 public:
706 // For each member of tuple 't', taken in order, evaluates '*out++ = f(t)'.
707 // Returns the final value of 'out' in case the caller needs it.
708 static OutIter Run(Func f, const Tuple& t, OutIter out) {
709 return IterateOverTuple<Tuple, TupleSize::value>()(f, t, out);
710 }
711
712 private:
713 template <typename Tup, size_t kRemainingSize>
714 struct IterateOverTuple {
715 OutIter operator() (Func f, const Tup& t, OutIter out) const {
716 *out++ = f(::std::get<TupleSize::value - kRemainingSize>(t));
717 return IterateOverTuple<Tup, kRemainingSize - 1>()(f, t, out);
718 }
719 };
720 template <typename Tup>
721 struct IterateOverTuple<Tup, 0> {
722 OutIter operator() (Func /* f */, const Tup& /* t */, OutIter out) const {
723 return out;
724 }
725 };
726};

732OutIter TransformTupleValues(Func f, const Tuple& t, OutIter out) {
733 return TransformTupleValuesHelper<Tuple, Func, OutIter>::Run(f, t, out);

869// String comparison for narrow or wide strings that can have embedded NUL
870// characters.
871template <typename StringType>
872bool CaseInsensitiveStringEquals(const StringType& s1,
873 const StringType& s2) {
874 // Are the heads equal?
875 if (!CaseInsensitiveCStringEquals(s1.c_str(), s2.c_str())) {
876 return false;
877 }
878
879 // Skip the equal heads.
880 const typename StringType::value_type nul = 0;
881 const size_t i1 = s1.find(nul), i2 = s2.find(nul);
882
883 // Are we at the end of either s1 or s2?
884 if (i1 == StringType::npos || i2 == StringType::npos) {
885 return i1 == i2;
886 }
887
888 // Are the tails equal?
889 return CaseInsensitiveStringEquals(s1.substr(i1 + 1), s2.substr(i2 + 1));
890}

1648// Implements the polymorphic floating point equality matcher, which matches
1649// two float values using ULP-based approximation or, optionally, a
1650// user-specified epsilon. The template is meant to be instantiated with
1651// FloatType being either float or double.
1652template <typename FloatType>
1653class FloatingEqMatcher {
1654 public:
1655 // Constructor for FloatingEqMatcher.
1656 // The matcher's input will be compared with expected. The matcher treats two
1657 // NANs as equal if nan_eq_nan is true. Otherwise, under IEEE standards,
1658 // equality comparisons between NANs will always return false. We specify a
1659 // negative max_abs_error_ term to indicate that ULP-based approximation will
1660 // be used for comparison.
1661 FloatingEqMatcher(FloatType expected, bool nan_eq_nan) :
1662 expected_(expected), nan_eq_nan_(nan_eq_nan), max_abs_error_(-1) {
1663 }
1664
1665 // Constructor that supports a user-specified max_abs_error that will be used
1666 // for comparison instead of ULP-based approximation. The max absolute
1667 // should be non-negative.
1668 FloatingEqMatcher(FloatType expected, bool nan_eq_nan,
1669 FloatType max_abs_error)
1670 : expected_(expected),
1671 nan_eq_nan_(nan_eq_nan),
1672 max_abs_error_(max_abs_error) {
1673 GTEST_CHECK_(max_abs_error >= 0)
1674 << ", where max_abs_error is" << max_abs_error;
1675 }

1677 // Implements floating point equality matcher as a Matcher<T>.
1678 template <typename T>
1679 class Impl : public MatcherInterface<T> {
1680 public:
1681 Impl(FloatType expected, bool nan_eq_nan, FloatType max_abs_error)
1682 : expected_(expected),
1683 nan_eq_nan_(nan_eq_nan),
1684 max_abs_error_(max_abs_error) {}
1685
1686 bool MatchAndExplain(T value,
1687 MatchResultListener* listener) const override {
1688 const FloatingPoint<FloatType> actual(value), expected(expected_);
1689
1690 // Compares NaNs first, if nan_eq_nan_ is true.
1691 if (actual.is_nan() || expected.is_nan()) {
1692 if (actual.is_nan() && expected.is_nan()) {
1693 return nan_eq_nan_;
1694 }
1695 // One is nan; the other is not nan.
1696 return false;
1697 }
1698 if (HasMaxAbsError()) {
1699 // We perform an equality check so that inf will match inf, regardless
1700 // of error bounds. If the result of value - expected_ would result in
1701 // overflow or if either value is inf, the default result is infinity,
1702 // which should only match if max_abs_error_ is also infinity.
1703 if (value == expected_) {
1704 return true;
1705 }
1706
1707 const FloatType diff = value - expected_;
1708 if (::std::fabs(diff) <= max_abs_error_) {
1709 return true;
1710 }
1711
1712 if (listener->IsInterested()) {
1713 *listener << "which is " << diff << " from " << expected_;
1714 }
1715 return false;
1716 } else {
1717 return actual.AlmostEquals(expected);
1718 }
1719 }
1720
1721 void DescribeTo(::std::ostream* os) const override {
1722 // os->precision() returns the previously set precision, which we
1723 // store to restore the ostream to its original configuration
1724 // after outputting.
1725 const ::std::streamsize old_precision = os->precision(
1726 ::std::numeric_limits<FloatType>::digits10 + 2);
1727 if (FloatingPoint<FloatType>(expected_).is_nan()) {
1728 if (nan_eq_nan_) {
1729 *os << "is NaN";
1730 } else {
1731 *os << "never matches";
1732 }
1733 } else {
1734 *os << "is approximately " << expected_;
1735 if (HasMaxAbsError()) {
1736 *os << " (absolute error <= " << max_abs_error_ << ")";
1737 }
1738 }
1739 os->precision(old_precision);
1740 }
1741
1742 void DescribeNegationTo(::std::ostream* os) const override {
1743 // As before, get original precision.
1744 const ::std::streamsize old_precision = os->precision(
1745 ::std::numeric_limits<FloatType>::digits10 + 2);
1746 if (FloatingPoint<FloatType>(expected_).is_nan()) {
1747 if (nan_eq_nan_) {
1748 *os << "isn't NaN";
1749 } else {
1750 *os << "is anything";
1751 }
1752 } else {
1753 *os << "isn't approximately " << expected_;
1754 if (HasMaxAbsError()) {
1755 *os << " (absolute error > " << max_abs_error_ << ")";
1756 }
1757 }
1758 // Restore original precision.
1759 os->precision(old_precision);
1760 }
1761
1762 private:
1763 bool HasMaxAbsError() const {
1764 return max_abs_error_ >= 0;
1765 }
1766
1767 const FloatType expected_;
1768 const bool nan_eq_nan_;
1769 // max_abs_error will be used for value comparison when >= 0.
1770 const FloatType max_abs_error_;
1771 };

2719 size_t i = 0;

2722 ++it, ++i) {

2752 *listener << " and";
2753 } else {

2755 }

3401 auto it = stl_container.begin();

3404

3406 // the end of either the elements or the matchers, or until we find a
3407 // mismatch.
3408 for (; it != stl_container.end() && exam_pos != count(); ++it, ++exam_pos) {
3409 bool match; // Does the current element match the current matcher?
3410 if (listener_interested) {

3426 // calling size() s.t. this code works for stream-like "containers"

3465 }
3466 }
3467 }

4732// container or a native array that contains the same number of
4733// elements as in rhs, where in some permutation of the container, its
4734// i-th element and rhs's i-th element (as a pair) satisfy the given

4736// cast to Matcher<std::tuple<const T1&, const T2&> >, where T1 and T2 are
4737// the types of elements in the LHS container and the RHS container
4738// respectively.

4740// This is like Pointwise(pair_matcher, rhs), except that the element
4741// order doesn't matter.

4743inline internal::UnorderedElementsAreArrayMatcher<
4744 typename internal::BoundSecondMatcher<
4745 Tuple2Matcher,
4746 typename internal::StlContainerView<
4747 typename std::remove_const<RhsContainer>::type>::type::value_type>>

4749 const RhsContainer& rhs_container) {

4751 // STL-style container and it being a native C-style array.
4752 typedef typename internal::StlContainerView<RhsContainer> RhsView;
4753 typedef typename RhsView::type RhsStlContainer;
4754 typedef typename RhsStlContainer::value_type Second;
4755 const RhsStlContainer& rhs_stl_container =
4756 RhsView::ConstReference(rhs_container);

4759 ::std::vector<internal::BoundSecondMatcher<Tuple2Matcher, Second> > matchers;
4760 for (auto it = rhs_stl_container.begin();
4761 it != rhs_stl_container.end(); ++it) {
4762 matchers.push_back(
4763 internal::MatcherBindSecond(tuple2_matcher, *it));
4764 }

4816// IsSupersetOf(iterator_first, iterator_last)
4817// IsSupersetOf(pointer, count)
4818// IsSupersetOf(array)
4819// IsSupersetOf(container)
4820// IsSupersetOf({e1, e2, ..., en})
4821//
4822// IsSupersetOf() verifies that a surjective partial mapping onto a collection
4823// of matchers exists. In other words, a container matches
4824// IsSupersetOf({e1, ..., en}) if and only if there is a permutation
4825// {y1, ..., yn} of some of the container's elements where y1 matches e1,
4826// ..., and yn matches en. Obviously, the size of the container must be >= n
4827// in order to have a match. Examples:
4828//
4829// - {1, 2, 3} matches IsSupersetOf({Ge(3), Ne(0)}), as 3 matches Ge(3) and

4832// both Eq(1) and Lt(2). The reason is that different matchers must be used
4833// for elements in different slots of the container.
4834// - {1, 1, 2} matches IsSupersetOf({Eq(1), Lt(2)}), as (the first) 1 matches
4835// Eq(1) and (the second) 1 matches Lt(2).
4836// - {1, 2, 3} matches IsSupersetOf(Gt(1), Gt(1)), as 2 matches (the first)
4837// Gt(1) and 3 matches (the second) Gt(1).
4838//
4839// The matchers can be specified as an array, a pointer and count, a container,
4840// an initializer list, or an STL iterator range. In each of these cases, the
4841// underlying matchers can be either values or matchers.

4850}
