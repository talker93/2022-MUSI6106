49template <typename Pattern>
50struct ThisRefAdjuster {
51 template <typename T>
52 using AdjustT = typename std::conditional<
53 std::is_const<typename std::remove_reference<Pattern>::type>::value,
54 typename std::conditional<std::is_lvalue_reference<Pattern>::value,
55 const T&, const T&&>::type,
56 typename std::conditional<std::is_lvalue_reference<Pattern>::value, T&,
57 T&&>::type>::type;
58
59 template <typename MockType>
60 static AdjustT<MockType> Adjust(const MockType& mock) {
61 return static_cast<AdjustT<MockType>>(const_cast<MockType&>(mock));
62 }
63};

65constexpr bool PrefixOf(const char* a, const char* b) {
66 return *a == 0 || (*a == *b && internal::PrefixOf(a + 1, b + 1));
67}
68
69template <int N, int M>
70constexpr bool StartsWith(const char (&prefix)[N], const char (&str)[M]) {
71 return N <= M && internal::PrefixOf(prefix, str);
72}
73
74template <int N, int M>
75constexpr bool EndsWith(const char (&suffix)[N], const char (&str)[M]) {
76 return N <= M && internal::PrefixOf(suffix, str + M - N);
77}
78
79template <int N, int M>
80constexpr bool Equals(const char (&a)[N], const char (&b)[M]) {
81 return N == M && internal::PrefixOf(a, b);
82}
83
84template <int N>
85constexpr bool ValidateSpec(const char (&spec)[N]) {
86 return internal::Equals("const", spec) ||
87 internal::Equals("override", spec) ||
88 internal::Equals("final", spec) ||
89 internal::Equals("noexcept", spec) ||
90 (internal::StartsWith("noexcept(", spec) &&
91 internal::EndsWith(")", spec)) ||
92 internal::Equals("ref(&)", spec) ||
93 internal::Equals("ref(&&)", spec) ||
94 (internal::StartsWith("Calltype(", spec) &&
95 internal::EndsWith(")", spec));
96}

170#define GMOCK_INTERNAL_MOCK_METHOD_IMPL(_N, _MethodName, _Constness, \
171 _Override, _Final, _NoexceptSpec, \
172 _CallType, _RefSpec, _Signature) \
173 typename ::testing::internal::Function<GMOCK_PP_REMOVE_PARENS( \
174 _Signature)>::Result \
175 GMOCK_INTERNAL_EXPAND(_CallType) \
176 _MethodName(GMOCK_PP_REPEAT(GMOCK_INTERNAL_PARAMETER, _Signature, _N)) \
177 GMOCK_PP_IF(_Constness, const, ) _RefSpec _NoexceptSpec \
178 GMOCK_PP_IF(_Override, override, ) GMOCK_PP_IF(_Final, final, ) { \
179 GMOCK_MOCKER_(_N, _Constness, _MethodName) \
180 .SetOwnerAndName(this, #_MethodName); \
181 return GMOCK_MOCKER_(_N, _Constness, _MethodName) \
182 .Invoke(GMOCK_PP_REPEAT(GMOCK_INTERNAL_FORWARD_ARG, _Signature, _N)); \
183 } \
184 ::testing::MockSpec<GMOCK_PP_REMOVE_PARENS(_Signature)> gmock_##_MethodName( \
185 GMOCK_PP_REPEAT(GMOCK_INTERNAL_MATCHER_PARAMETER, _Signature, _N)) \
186 GMOCK_PP_IF(_Constness, const, ) _RefSpec { \
187 GMOCK_MOCKER_(_N, _Constness, _MethodName).RegisterOwner(this); \
188 return GMOCK_MOCKER_(_N, _Constness, _MethodName) \
189 .With(GMOCK_PP_REPEAT(GMOCK_INTERNAL_MATCHER_ARGUMENT, , _N)); \
190 } \
191 ::testing::MockSpec<GMOCK_PP_REMOVE_PARENS(_Signature)> gmock_##_MethodName( \
192 const ::testing::internal::WithoutMatchers&, \
193 GMOCK_PP_IF(_Constness, const, )::testing::internal::Function< \
194 GMOCK_PP_REMOVE_PARENS(_Signature)>*) const _RefSpec _NoexceptSpec { \
195 return ::testing::internal::ThisRefAdjuster<GMOCK_PP_IF( \
196 _Constness, const, ) int _RefSpec>::Adjust(*this) \
197 .gmock_##_MethodName(GMOCK_PP_REPEAT( \
198 GMOCK_INTERNAL_A_MATCHER_ARGUMENT, _Signature, _N)); \
199 } \
200 mutable ::testing::FunctionMocker<GMOCK_PP_REMOVE_PARENS(_Signature)> \
201 GMOCK_MOCKER_(_N, _Constness, _MethodName)

294// Note: The use of `identity_t` here allows _Ret to represent return types that
295// would normally need to be specified in a different way. For example, a method
296// returning a function pointer must be written as
297//
298// fn_ptr_return_t (*method(method_args_t...))(fn_ptr_args_t...)
299//
300// But we only support placing the return type at the beginning. To handle this,
301// we wrap all calls in identity_t, so that a declaration will be expanded to
302//
303// identity_t<fn_ptr_return_t (*)(fn_ptr_args_t...)> method(method_args_t...)
304//
305// This allows us to work around the syntactic oddities of function/method
306// types.
307#define GMOCK_INTERNAL_SIGNATURE(_Ret, _Args) \
308 ::testing::internal::identity_t<GMOCK_PP_IF(GMOCK_PP_IS_BEGIN_PARENS(_Ret), \
309 GMOCK_PP_REMOVE_PARENS, \
310 GMOCK_PP_IDENTITY)(_Ret)>( \
311 GMOCK_PP_FOR_EACH(GMOCK_INTERNAL_GET_TYPE, _, _Args))
312
313#define GMOCK_INTERNAL_GET_TYPE(_i, _, _elem) \
314 GMOCK_PP_COMMA_IF(_i) \
315 GMOCK_PP_IF(GMOCK_PP_IS_BEGIN_PARENS(_elem), GMOCK_PP_REMOVE_PARENS, \
316 GMOCK_PP_IDENTITY) \
317 (_elem)
318
319#define GMOCK_INTERNAL_PARAMETER(_i, _Signature, _) \
320 GMOCK_PP_COMMA_IF(_i) \
321 GMOCK_INTERNAL_ARG_O(_i, GMOCK_PP_REMOVE_PARENS(_Signature)) \
322 gmock_a##_i
323
324#define GMOCK_INTERNAL_FORWARD_ARG(_i, _Signature, _) \
325 GMOCK_PP_COMMA_IF(_i) \
326 ::std::forward<GMOCK_INTERNAL_ARG_O( \
327 _i, GMOCK_PP_REMOVE_PARENS(_Signature))>(gmock_a##_i)
328
329#define GMOCK_INTERNAL_MATCHER_PARAMETER(_i, _Signature, _) \
330 GMOCK_PP_COMMA_IF(_i) \
331 GMOCK_INTERNAL_MATCHER_O(_i, GMOCK_PP_REMOVE_PARENS(_Signature)) \
332 gmock_a##_i
333
334#define GMOCK_INTERNAL_MATCHER_ARGUMENT(_i, _1, _2) \
335 GMOCK_PP_COMMA_IF(_i) \
336 gmock_a##_i
337
338#define GMOCK_INTERNAL_A_MATCHER_ARGUMENT(_i, _Signature, _) \
339 GMOCK_PP_COMMA_IF(_i) \
340 ::testing::A<GMOCK_INTERNAL_ARG_O(_i, GMOCK_PP_REMOVE_PARENS(_Signature))>()
