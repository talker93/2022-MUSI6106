99

133Error_t CVibrato::setParam( VibratoParam_t eParam, float fParamValue )
134{
135 if (!m_bIsInitialized)
136 return Error_t::kNotInitializedError;
137 if (!isInParamRange(eParam, fParamValue))
138 return Error_t::kFunctionInvalidArgsError;
139
140 switch (eParam)
141 {
142 case kParamModFreqInHz:
143 return m_pCLfo->setParam(CLfo::kLfoParamFrequency, fParamValue);
144 case kParamModWidthInS:
145 return m_pCLfo->setParam(CLfo::kLfoParamAmplitude, fParamValue * m_fSampleRate);
146 case kNumVibratoParams:
147 return Error_t::kFunctionInvalidArgsError;
148 }
149
150 return Error_t::kNoError;
151}

178Error_t CVibrato::process( float **ppfInputBuffer, float **ppfOutputBuffer, int iNumberOfFrames )
179{
180 if (!ppfInputBuffer || !ppfOutputBuffer || iNumberOfFrames < 0)
181 return Error_t::kFunctionInvalidArgsError;
182
183 for (int i = 0; i < iNumberOfFrames; i++)
184 {
185 float fOffset = m_pCLfo->getNext();
186 for (int c = 0; c < m_iNumChannels; c++)
187 {
188 m_ppCRingBuff[c]->putPostInc(ppfInputBuffer[c][i]);
189
190 ppfOutputBuffer[c][i] = m_ppCRingBuff[c]->get(fOffset);
191 m_ppCRingBuff[c]->getPostInc(); // dummy call to keep write and read idx in sync
192 }
193 }
194
195 return Error_t::kNoError;
196}
