46Error_t CFft::initInstance( int iBlockLength, int iZeroPadFactor, WindowFunction_t eWindow /*= kWindowHann*/, Windowing_t eWindowing /*= kPreWindow*/ )
47{
48 Error_t rErr = Error_t::kNoError;
49
50 // sanity check
51 if (!CUtil::isPowOf2(iBlockLength) || iZeroPadFactor <= 0 || !CUtil::isPowOf2(iBlockLength*iZeroPadFactor))
52 return Error_t::kFunctionInvalidArgsError;
53
54 // clean up
55 resetInstance();
56
57 m_iDataLength = iBlockLength;
58 m_iFftLength = iBlockLength * iZeroPadFactor;
59
60 m_ePrePostWindowOpt = eWindowing;
61
62 rErr = allocMemory ();
63 if (rErr != Error_t::kNoError)
64 return rErr;
65
66 rErr = computeWindow (eWindow);
67
68 m_bIsInitialized = true;
69
70 return rErr;
71}

111Error_t CFft::doFft( complex_t *pfSpectrum, const float *pfInput )
112{
113 if (!m_bIsInitialized)
114 return Error_t::kNotInitializedError;
115 if (!pfInput || !pfSpectrum)
116 return Error_t::kFunctionInvalidArgsError;
117
118 // copy data to internal buffer
119 CVectorFloat::copy(m_pfProcessBuff, pfInput, m_iDataLength);
120 CVectorFloat::setZero(&m_pfProcessBuff[m_iDataLength], m_iFftLength-m_iDataLength);
121
122 // apply window function
123 if (m_ePrePostWindowOpt & kPreWindow)
124 CVectorFloat::mul_I(m_pfProcessBuff, m_pfWindowBuff, m_iDataLength);
125
126 // compute fft
127 LaszloFft::realfft_split(m_pfProcessBuff, m_iFftLength);
128
129 // copy data to output buffer
130 CVectorFloat::copy(pfSpectrum, m_pfProcessBuff, m_iFftLength);
131
132 return Error_t::kNoError;
133}

156Error_t CFft::getMagnitude( float *pfMag, const complex_t *pfSpectrum ) const
157{
158 if (!m_bIsInitialized)
159 return Error_t::kNotInitializedError;
160
161 // re(0),re(1),re(2),...,re(size/2),im(size/2-1),...,im(1)
162 int iNyq = m_iFftLength>>1;
163
164 // no imaginary part at these bins
165 pfMag[0] = std::abs(pfSpectrum[0]);
166 pfMag[iNyq] = std::abs(pfSpectrum[iNyq]);
167
168 for (int i = 1; i < iNyq; i++)
169 {
170 int iImagIdx = m_iFftLength - i;
171 pfMag[i] = sqrtf(pfSpectrum[i]*pfSpectrum[i] + pfSpectrum[iImagIdx]*pfSpectrum[iImagIdx]);
172 }
173 return Error_t::kNoError;
174}

176Error_t CFft::getPhase( float *pfPhase, const complex_t *pfSpectrum ) const
177{
178 if (!m_bIsInitialized)
179 return Error_t::kNotInitializedError;
180
181 // re(0),re(1),re(2),...,re(size/2),im(size/2-1),...,im(1)
182 int iNyq = m_iFftLength>>1;
183
184 pfPhase[0] = m_Pi;
185 pfPhase[iNyq] = m_Pi;
186
187 for (int i = 1; i < iNyq; i++)
188 {
189 int iImagIdx = m_iFftLength - i;
190 if (pfSpectrum[i] == .0F && pfSpectrum[iImagIdx] != .0F)
191 pfPhase[i] = m_Pi2;
192 else
193 pfPhase[i] = atan2f (pfSpectrum[iImagIdx], pfSpectrum[i]);
194 }
195 return Error_t::kNoError;

270Error_t CFft::computeWindow( WindowFunction_t eWindow )
271{
272 int i;
273
274 // note that these windows are periodic, not symmetric
275 switch (eWindow)
276 {
277 case kWindowSine:
278 {
279 for (i = 0; i < m_iDataLength; i++)
280 {
281 m_pfWindowBuff[i] = sinf((i * m_Pi) / (m_iDataLength+1));
282 }
283 break;
284 }
285 case kWindowHann:
286 {
287 for (i = 0; i < m_iDataLength; i++)
288 {
289 m_pfWindowBuff[i] = .5F * (1.F - cosf((i * 2.F*m_Pi) / (m_iDataLength+1)));
290 }
291 break;
292 }
293 case kWindowHamming:
294 {
295 for (i = 0; i < m_iDataLength; i++)
296 {
297 m_pfWindowBuff[i] = .54F - .46F * cosf((i * 2.F*m_Pi) / (m_iDataLength+1));
298 }
299 break;
300 }
301 }
302
303 return Error_t::kNoError;
304}
