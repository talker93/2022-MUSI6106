76// GetRawPointer(p) returns the raw pointer underlying p when p is a
77// smart pointer, or returns p itself when p is already a raw pointer.
78// The following default implementation is for the smart pointer case.

104// In what follows, we use the term "kind" to indicate whether a type
105// is bool, an integer type (excluding bool), a floating-point type,
106// or none of them. This categorization is useful for determining
107// when a matcher argument type can be safely converted to another
108// type in the implementation of SafeMatcherCast.
109enum TypeKind {
110 kBool, kInteger, kFloatingPoint, kOther

153// LosslessArithmeticConvertibleImpl<kFromKind, From, kToKind, To>::value
154// is true if and only if arithmetic type From can be losslessly converted to
155// arithmetic type To.
156//
157// It's the user's responsibility to ensure that both From and To are
158// raw (i.e. has no CV modifier, is not a pointer, and is not a
159// reference) built-in arithmetic types, kFromKind is the kind of
160// From, and kToKind is the kind of To; the value is
161// implementation-defined when the above pre-condition is violated.
162template <TypeKind kFromKind, typename From, TypeKind kToKind, typename To>
163using LosslessArithmeticConvertibleImpl = std::integral_constant<
164 bool,
165 // clang-format off
166 // Converting from bool is always lossless
167 (kFromKind == kBool) ? true
168 // Converting between any other type kinds will be lossy if the type
169 // kinds are not the same.
170 : (kFromKind != kToKind) ? false
171 : (kFromKind == kInteger &&
172 // Converting between integers of different widths is allowed so long
173 // as the conversion does not go from signed to unsigned.
174 (((sizeof(From) < sizeof(To)) &&
175 !(std::is_signed<From>::value && !std::is_signed<To>::value)) ||
176 // Converting between integers of the same width only requires the
177 // two types to have the same signedness.
178 ((sizeof(From) == sizeof(To)) &&
179 (std::is_signed<From>::value == std::is_signed<To>::value)))
180 ) ? true
181 // Floating point conversions are lossless if and only if `To` is at least
182 // as wide as `From`.
183 : (kFromKind == kFloatingPoint && (sizeof(From) <= sizeof(To))) ? true
184 : false
185 // clang-format on
186 >;
187
188// LosslessArithmeticConvertible<From, To>::value is true if and only if
189// arithmetic type From can be losslessly converted to arithmetic type To.
190//
191// It's the user's responsibility to ensure that both From and To are
192// raw (i.e. has no CV modifier, is not a pointer, and is not a
193// reference) built-in arithmetic types; the value is
194// implementation-defined when the above pre-condition is violated.
195template <typename From, typename To>
196using LosslessArithmeticConvertible =
197 LosslessArithmeticConvertibleImpl<GMOCK_KIND_OF_(From), From,
198 GMOCK_KIND_OF_(To), To>;

219// Asserts that condition is true; aborts the process with the given
220// message if condition is false. We cannot use LOG(FATAL) or CHECK()
221// as Google Mock might be used to mock the log sink itself. We
222// inline this function to prevent it from showing up in the stack
223// trace.
224inline void Assert(bool condition, const char* file, int line,
225 const std::string& msg) {
226 if (!condition) {
227 GetFailureReporter()->ReportFailure(FailureReporterInterface::kFatal,
228 file, line, msg);
229 }
230}
231inline void Assert(bool condition, const char* file, int line) {
232 Assert(condition, file, line, "Assertion failed.");

299// Invalid<T>() is usable as an expression of type T, but will terminate
300// the program with an assertion failure if actually run. This is useful
301// when a value of type T is needed for compilation, but the statement
302// will not really be executed (or we don't care if the statement
303// crashes).
304template <typename T>
305inline T Invalid() {
306 Assert(false, "", -1, "Internal error: attempt to return invalid value");
307 // This statement is unreachable, and would never terminate even if it
308 // could be reached. It is provided only to placate compiler warnings
309 // about missing return statements.
310 return Invalid<T>();
311}

317// Given a raw type (i.e. having no top-level reference or const
318// modifier) RawContainer that's either an STL-style container or a
319// native array, class StlContainerView<RawContainer> has the
320// following members:
321//
322// - type is a type that provides an STL-style container view to
323// (i.e. implements the STL container concept for) RawContainer;
324// - const_reference is a type that provides a reference to a const
325// RawContainer;
326// - ConstReference(raw_container) returns a const reference to an STL-style
327// container view to raw_container, which is a RawContainer.
328// - Copy(raw_container) returns an STL-style container view of a
329// copy of raw_container, which is a RawContainer.
330//
331// This generic version is used when RawContainer itself is already an
332// STL-style container.
333template <class RawContainer>
334class StlContainerView {
335 public:
336 typedef RawContainer type;
337 typedef const type& const_reference;
338
339 static const_reference ConstReference(const RawContainer& container) {
340 static_assert(!std::is_const<RawContainer>::value,
341 "RawContainer type must not be const");
342 return container;
343 }
344 static type Copy(const RawContainer& container) { return container; }
345};
346
347// This specialization is used when RawContainer is a native array type.
348template <typename Element, size_t N>
349class StlContainerView<Element[N]> {
350 public:
351 typedef typename std::remove_const<Element>::type RawElement;
352 typedef internal::NativeArray<RawElement> type;
353 // NativeArray<T> can represent a native array either by value or by
354 // reference (selected by a constructor argument), so 'const type'
355 // can be used to reference a const native array. We cannot
356 // 'typedef const type& const_reference' here, as that would mean
357 // ConstReference() has to return a reference to a local variable.
358 typedef const type const_reference;
359
360 static const_reference ConstReference(const Element (&array)[N]) {
361 static_assert(std::is_same<Element, RawElement>::value,
362 "Element type must not be const");
363 return type(array, N, RelationToSourceReference());
364 }
365 static type Copy(const Element (&array)[N]) {
366 return type(array, N, RelationToSourceCopy());
367 }
368};
369
370// This specialization is used when RawContainer is a native array
371// represented as a (pointer, size) tuple.
372template <typename ElementPointer, typename Size>
373class StlContainerView< ::std::tuple<ElementPointer, Size> > {
374 public:
375 typedef typename std::remove_const<
376 typename std::pointer_traits<ElementPointer>::element_type>::type
377 RawElement;
378 typedef internal::NativeArray<RawElement> type;
379 typedef const type const_reference;
380
381 static const_reference ConstReference(
382 const ::std::tuple<ElementPointer, Size>& array) {
383 return type(std::get<0>(array), std::get<1>(array),
384 RelationToSourceReference());
385 }
386 static type Copy(const ::std::tuple<ElementPointer, Size>& array) {
387 return type(std::get<0>(array), std::get<1>(array), RelationToSourceCopy());
388 }
389};

430// Template struct Function<F>, where F must be a function type, contains
431// the following typedefs:
432//
433// Result: the function's return type.
434// Arg<N>: the type of the N-th argument, where N starts with 0.
435// ArgumentTuple: the tuple type consisting of all parameters of F.
436// ArgumentMatcherTuple: the tuple type consisting of Matchers for all
437// parameters of F.
438// MakeResultVoid: the function type obtained by substituting void
439// for the return type of F.
440// MakeResultIgnoredValue:
441// the function type obtained by substituting Something
442// for the return type of F.
443template <typename T>
444struct Function;
445
446template <typename R, typename... Args>
447struct Function<R(Args...)> {
448 using Result = R;
449 static constexpr size_t ArgumentCount = sizeof...(Args);
450 template <size_t I>
451 using Arg = ElemFromList<I, Args...>;
452 using ArgumentTuple = std::tuple<Args...>;
453 using ArgumentMatcherTuple = std::tuple<Matcher<Args>...>;
454 using MakeResultVoid = void(Args...);
455 using MakeResultIgnoredValue = IgnoredValue(Args...);
456};
457
458template <typename R, typename... Args>
459constexpr size_t Function<R(Args...)>::ArgumentCount;
